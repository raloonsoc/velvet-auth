import { Elysia } from "elysia"
import { velvetAuth, createAuthGuard } from "velvet-auth"

const app = new Elysia()
  .use(velvetAuth(userStore, email, {
    jwt:      { secret: process.env.JWT_SECRET },
    session:  { store: redis, ttl: "7d" },
    password: "argon2id",
  }))
  .use(createAuthGuard())
  .get("/me", ({ user }) => user)
